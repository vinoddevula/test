DECLARE
   ltab_Compare_Activity_Table   forms_utilities.Compare_Activity_Table;
   ln_count                      NUMBER := 0;
   ls_prev_airport_ident         VARCHAR2 (30) := '$$$$';
   ls_SSA												 VARCHAR2 (20) := NULL;
BEGIN
		IF :COMPARE_ACT_CYCL.DATA_SUPPLIER IS NULL THEN
				dsp_msg('Select Data Supplier.');
				GO_ITEM('COMPARE_ACT_CYCL.DATA_SUPPLIER');
				Raise Form_Trigger_Failure;
		END IF;		
				
		IF :COMPARE_ACT_CYCL.CURR_CYCLE IS NULL THEN
				dsp_msg('Select Current Cycle.');
				GO_ITEM('COMPARE_ACT_CYCL.CURR_CYCLE');
				Raise Form_Trigger_Failure;
		END IF;		
	
		IF :COMPARE_ACT_CYCL.PREV_CYCLE IS NULL THEN
				dsp_msg('Select Previous Cycle.');
				GO_ITEM('COMPARE_ACT_CYCL.PREV_CYCLE');
				Raise Form_Trigger_Failure;
		END IF;		

	  IF :COMPARE_ACT_CYCL.SID = 'D' THEN
	     ls_SSA := ls_SSA||:COMPARE_ACT_CYCL.SID;
	  END IF;
	  IF :COMPARE_ACT_CYCL.STAR = 'E' THEN
	     ls_SSA := ls_SSA||:COMPARE_ACT_CYCL.STAR;
	  END IF;
	  IF :COMPARE_ACT_CYCL.APPROACH = 'F' THEN
	     ls_SSA := ls_SSA||:COMPARE_ACT_CYCL.APPROACH;
	  END IF;
	
   GO_BLOCK ('SSA');
   CLEAR_BLOCK (NO_COMMIT);
   forms_utilities.Compare_Activity_Report (ls_SSA,
                                            :COMPARE_ACT_CYCL.DATA_SUPPLIER,
                                            :COMPARE_ACT_CYCL.CURR_CYCLE,
                                            :COMPARE_ACT_CYCL.PREV_CYCLE,
                                            :COMPARE_ACT_CYCL.DEL_ADD,
                                            ln_count,
                                            ltab_COMPARE_ACTIVITY_TABLE,
                                            :COMPARE_ACT_CYCL.AIRPORT_IDENT,
                                            :COMPARE_ACT_CYCL.ARPT_RANGE,
                                            NULL
                                            );
   :COMPARE_ACT_CYCL.TOTAL_CNT := ln_count;

   IF ln_count > 0 THEN
      GO_BLOCK ('SSA');
      FOR i IN ltab_Compare_Activity_Table.FIRST .. ltab_Compare_Activity_Table.LAST LOOP
         IF ls_prev_airport_ident <> ltab_Compare_Activity_Table (i).AIRPORT_IDENT THEN
            :SSA.AIRPORT_IDENT := ltab_Compare_Activity_Table (i).AIRPORT_IDENT;
         END IF;

         :SSA.APPROACH_IDENT 	:= ltab_Compare_Activity_Table (i).APPROACH_IDENT;
				 :SSA.PROC_TYPE 		 	:= ltab_Compare_Activity_Table (i).PROC_CODE;
				 	
					  IF ltab_Compare_Activity_Table (i).ACTIVITY = 'ON_ADD' THEN
					  	:SSA.ACTIVITY 	:= 'ADD';
					  ELSIF ltab_Compare_Activity_Table (i).ACTIVITY = 'ON_DEL' THEN	
					  	:SSA.ACTIVITY 	:= 'DEL';
					  ELSIF ltab_Compare_Activity_Table (i).ACTIVITY = 'APT_ADD' THEN
							:SSA.ACTIVITY 	:= ltab_COMPARE_ACTIVITY_TABLE (i).ACTIVITY;	
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.AIRPORT_IDENT',	current_record, visual_attribute,'APT_ADD');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.APPROACH_IDENT',current_record, visual_attribute,'APT_ADD');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.PROC_TYPE',			current_record, visual_attribute,'APT_ADD');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.ACTIVITY',			current_record, visual_attribute,'APT_ADD');
            ELSIF ltab_Compare_Activity_Table (i).ACTIVITY = 'APT_DEL' THEN
            	:SSA.ACTIVITY 	:= ltab_Compare_Activity_Table (i).ACTIVITY;	
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.AIRPORT_IDENT',	current_record, visual_attribute,'APT_DEL');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.APPROACH_IDENT',current_record, visual_attribute,'APT_DEL');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.PROC_TYPE',			current_record, visual_attribute,'APT_DEL');
	      			SET_ITEM_INSTANCE_PROPERTY('SSA.ACTIVITY',			current_record, visual_attribute,'APT_DEL');
            END IF;
         ls_prev_airport_ident := ltab_Compare_Activity_Table (i).AIRPORT_IDENT;
         NEXT_RECORD;
      END LOOP;
      FIRST_RECORD;
   ELSE
   	dsp_msg ('None of the Procedures are available.');
    GO_BLOCK ('COMPARE_ACT_CYCL');
   END IF;
END;

   ls_SSA                        VARCHAR2 (20) := NULL;
BEGIN

   IF :COMPARE_ACT_SUPP.PROCESSING_CYCLE IS NULL THEN

      LOOP
         :SSA_SUPP.APPROACH_IDENT 	:= ltab_Compare_Activity_Table (i).APPROACH_IDENT;
         :SSA_SUPP.PROC_TYPE 				:= ltab_Compare_Activity_Table (i).PROC_CODE;

         IF ls_Prev_Airport_Ident <> ltab_Compare_Activity_Table (i).AIRPORT_IDENT THEN
            :SSA_SUPP.AIRPORT_IDENT := ltab_Compare_Activity_Table (i).AIRPORT_IDENT;
         END IF;

         IF ltab_Compare_Activity_Table (i).ACTIVITY = 'ON_ADD' THEN
            :SSA_SUPP.ACTIVITY 			:= 'ADD_IN_' || :COMPARE_ACT_SUPP.TO_SUPPLIER;
         ELSIF ltab_Compare_Activity_Table (i).ACTIVITY = 'ON_DEL' THEN
            :SSA_SUPP.ACTIVITY 			:= 'DEL_IN_' || :COMPARE_ACT_SUPP.FROM_SUPPLIER;
         ELSIF ltab_Compare_Activity_Table (i).ACTIVITY LIKE 'APT_ADD' THEN
            :SSA_SUPP.ACTIVITY 			:= 'APT_ADD_IN_' || :COMPARE_ACT_SUPP.TO_SUPPLIER;
            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.AIRPORT_IDENT', 	current_record, visual_attribute, 'APT_ADD');
            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.APPROACH_IDENT', 	current_record, visual_attribute, 'APT_ADD');
            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.PROC_TYPE', 			current_record, visual_attribute, 'APT_ADD');
            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.ACTIVITY', 				current_record, visual_attribute, 'APT_ADD');

            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.PROC_TYPE', 			current_record, visual_attribute, 'APT_DEL');
            SET_ITEM_INSTANCE_PROPERTY ('SSA_SUPP.ACTIVITY', 				current_record, visual_attribute, 'APT_DEL');
         END IF;

         ls_Prev_Airport_Ident 			:= ltab_Compare_Activity_Table (i).AIRPORT_IDENT;
         NEXT_RECORD;
      END LOOP;

      FIRST_RECORD;
   ELSE
      dsp_msg ('None of the Procedures are available.');
      GO_BLOCK ('COMPARE_ACT_SUPP');
   END IF;

DECLARE
   TYPE Type_Ref_Cur IS REF CURSOR;
   tld_arpt_cur      Type_Ref_Cur;
   ls_Airport_Ident  pl_tld_airport.Airport_Ident%TYPE;
BEGIN
   GO_BLOCK ('COMPARE_ACT_CUST');
   CLEAR_BLOCK (NO_COMMIT);	
   :TLD_STD_AIRPORTS.TOTAL_CUSTOMERS := 0;
   --
   IF :TLD_STD_AIRPORTS.PROCESSING_CYCLE IS NULL THEN
      dsp_msg ('Select Processing Cycle.');
      GO_ITEM ('TLD_STD_AIRPORTS.PROCESSING_CYCLE');
      RAISE Form_Trigger_Failure;
   END IF;
   IF :TLD_STD_AIRPORTS.DATA_SUPPLIER IS NULL THEN
      dsp_msg ('Select Data Supplier.');
      GO_ITEM ('TLD_STD_AIRPORTS.DATA_SUPPLIER');
      RAISE Form_Trigger_Failure;
   END IF;
   --
   GO_BLOCK ('COMPARE_ACT_ARPT');
   CLEAR_BLOCK (NO_COMMIT);

   OPEN tld_arpt_cur FOR
		   SELECT tld_apt.airport_ident tld_arpt
		     FROM pl_tld_airport tld_apt, 
		     	    pl_std_airport std_apt
		    WHERE tld_apt.data_supplier     = 	std_apt.data_supplier
		      AND tld_apt.processing_cycle  = 	std_apt.processing_cycle
		      AND tld_apt.airport_ident 		= 	std_apt.airport_ident
		      AND tld_apt.processing_cycle 	= 	:TLD_STD_AIRPORTS.PROCESSING_CYCLE
		      AND tld_apt.data_supplier 		= 	'' || :TLD_STD_AIRPORTS.DATA_SUPPLIER || ''
		      AND tld_apt.airport_ident 	LIKE 	'' || NVL(:TLD_STD_AIRPORTS.AIRPORT_IDENT,'%') || ''
	   GROUP BY tld_apt.airport_ident
	   ORDER BY 1;

   IF :TLD_STD_AIRPORTS.PROCESSING_CYCLE IS NULL THEN
      dsp_msg ('Select Processing Cycle.');

   OPEN tld_arpt_cur FOR
       SELECT tld_apt.airport_ident tld_arpt, tld_apt.customer_ident cust_arpt
         FROM pl_tld_airport tld_apt, 
         			pl_std_airport std_apt
        WHERE tld_apt.data_supplier     = 	std_apt.data_supplier

     GROUP BY tld_apt.airport_ident, tld_apt.customer_ident
     ORDER BY 1, 2;

      FETCH tld_arpt_cur INTO :COMPARE_ACT_CUST.AIRPORT_IDENT, :COMPARE_ACT_CUST.CUST_IDENT;

   	GO_BLOCK('TLD_STD_AIRPORTS');
   END IF;	
   CLOSE tld_arpt_cur;
   FIRST_RECORD;

DECLARE
	    cursor rec_cnt(proc_cycle number,prev_cycle NUMBER,supplier varchar2) is
        select rc1.table_name,rc1.data_supplier,rc1.new_cycle as CYCLE1,rc1.records_changed+rc1.records_added+rc1.records_unchanged as TOTAL_REC1,
        rc2.new_cycle as CYCLE2,rc2.records_changed+rc2.records_added+rc2.records_unchanged as TOTAL_REC2,
        rc1.records_changed,rc1.records_deleted,rc1.records_added,rc1.records_unchanged 
        from cptm_record_count rc1, cptm_record_count rc2
        where rc1.new_cycle = proc_cycle
        and rc2.new_cycle = prev_cycle
        and rc1.data_supplier = supplier
        and rc1.data_supplier = rc2.data_supplier
        and rc1.table_name = rc2.table_name
        and rc1.table_name like '%STD%'
        and rc2.table_name like '%STD%'
        order by rc1.table_name;
        
   ls_curr_cycle  cptm_record_count.new_cycle%TYPE;
   ls_prev_cycle  cptm_record_count.new_cycle%TYPE;
   cnt NUMBER:=0;
BEGIN
	--ls_curr_cycle := :STD_RECORD_;
	GO_BLOCK ('COMPARE_RECORD_COUNT_BLK');
  CLEAR_BLOCK (NO_COMMIT);
  
  IF :STD_RECORD_COUNT_BLK.PROCESSING_CYCLE IS NULL THEN
      dsp_msg ('Select Processing Cycle.');
      GO_ITEM ('STD_RECORD_COUNT_BLK.PROCESSING_CYCLE');
      RAISE Form_Trigger_Failure;
   END IF;
   IF :STD_RECORD_COUNT_BLK.DATA_SUPPLIER IS NULL THEN
      dsp_msg ('Select Data Supplier.');
      GO_ITEM ('STD_RECORD_COUNT_BLK.DATA_SUPPLIER');
      RAISE Form_Trigger_Failure;
   END IF;
  	
	FOR rec IN rec_cnt(:STD_RECORD_COUNT_BLK.PROCESSING_CYCLE,util2.get_previous_cycle(:STD_RECORD_COUNT_BLK.PROCESSING_CYCLE),
											:STD_RECORD_COUNT_BLK.DATA_SUPPLIER) 
	LOOP
		
		:COMPARE_RECORD_COUNT_BLK.DATA_TYPE := REC.table_name;
		:COMPARE_RECORD_COUNT_BLK.SUPPLIER := REC.data_supplier;
		:COMPARE_RECORD_COUNT_BLK.COUNT1 := REC.TOTAL_REC2;
		:COMPARE_RECORD_COUNT_BLK.COUNT2 := REC.TOTAL_REC1;
		:COMPARE_RECORD_COUNT_BLK.ADDED := REC.records_added;
		:COMPARE_RECORD_COUNT_BLK.DELETED := REC.records_deleted;
		:COMPARE_RECORD_COUNT_BLK.CHANGED := REC.records_changed;
		:COMPARE_RECORD_COUNT_BLK.UNCHANGED := REC.records_unchanged;
		cnt:=cnt+1;
		NEXT_RECORD;
		
	END LOOP;
	FIRST_RECORD;
	IF cnt=0 THEN
		dsp_msg('No Standard Data exists for Selected Cycle');
		RAISE Form_Trigger_Failure;
	END IF;
	
END;

  FUNCTION IsBlockVisible (BlockName IN VARCHAR2) RETURN BOOLEAN
  IS
    Result     BOOLEAN := false;
    ItemName   VARCHAR2(65);
  BEGIN
  	--break;
    ItemName := BlockName||'.'||GET_BLOCK_PROPERTY(BlockName, FIRST_ITEM);
    WHILE ItemName IS NOT NULL and Result = FALSE
    LOOP
    	if GET_ITEM_PROPERTY(ItemName, VISIBLE) = 'TRUE' then
        Result := TRUE;
      else
        ItemName := GET_ITEM_PROPERTY(ItemName, NEXTITEM);
        if ItemName IS NOT NULL then
          ItemName := BlockName||'.'||ItemName;
      	end if;
      end if;
    end loop;	
    RETURN(Result);
  END IsBlockVisible; --    IsBlockVisible 

  -- Determine if an ITEM that can contain data is visible (displayed).
  FUNCTION IsDataItemVisible (ItemId  IN item,

  IS

  --
  -- Output the header for a data block in delimited format.
  -- 
  PROCEDURE WriteBlockHeaderDelimited( pFileId    in Text_IO.file_type,
                                       Delimiter IN VARCHAR2,
                                       BlockName IN VARCHAR2,
                                       TabPage   IN VARCHAR2,
                                       Indent    IN NUMBER)
  IS
  	Heading     VARCHAR2(300);
    ItemName		VARCHAR2(65);
    ItemPrompt	VARCHAR2(40);
    ItemId			Item;
    i           NUMBER(2);
  	ItemType    VARCHAR2(20);
  	Relation    VARCHAR2(40);
  	DetailRelation VARCHAR2(40);
  	DetailBlock VARCHAR2(40);
  	ContextSaved BOOLEAN;
  	block_id 		BLOCK;
  Begin
    if IsBlockVisible(BlockName) = FALSE then return; end if;
  	ItemName := BlockName||'.'||GET_BLOCK_PROPERTY(BlockName, FIRST_ITEM);
  	ItemId := FIND_ITEM(ItemName);
  	-- Indentation
  	Text_IO.PUT_LINE(pFileId,' ');
    for i in 1..Indent loop
    	Text_IO.PUT(pFileId, Delimiter); -- use delimiter to indent
    end loop;
    -- Display each item that is visible, as a column heading
  	WHILE ItemName IS NOT NULL   LOOP
      if IsDataItemVisible( ItemId, TabPage ) = TRUE then
 		    if ItemName not like 'DETAIL.%' then
 		    	Heading := GET_ITEM_PROPERTY(ItemId, PROMPT_TEXT); -- Use Item Prompt
 		    	IF Heading IS NULL then
	    	  	ItemType := GET_ITEM_PROPERTY(ItemId, ITEM_TYPE);
  		    	IF ItemType = 'CHECKBOX' OR
  		    		 ItemType = 'BUTTON'   OR
  		    	 	ItemType = 'RADIOGROUP' THEN
  		      	Heading := GET_ITEM_PROPERTY(ItemId, LABEL);   -- Use Item Label
  		    	end if;
  		    	if Heading IS NULL then
  		      	Heading := replace( ItemName, BlockName||'.'); -- Use Item Name(Default)
  		    	end if;
  		    end if;	
  		  else
  		  	Heading := substr(ItemName,8);
  		  end if; 
  		  Text_IO.PUT(pFileId, '"'||Heading||'"'||Delimiter);
      END IF;
   		ItemName := GET_ITEM_PROPERTY(ItemId, NEXTITEM);
   		IF ItemName IS NOT NULL THEN
        ItemName := BlockName||'.'||ItemName;
    	  ItemId := FIND_ITEM(ItemName);
    	END IF;
  	END LOOP; -- WHILE
    	-- Get the Master-Detail relationships for the record
    	block_id := FIND_BLOCK(BlockName);
    	Relation := GET_BLOCK_PROPERTY(block_id, FIRST_MASTER_RELATION);
    	while Relation is not NULL loop
    		DetailBlock := GET_RELATION_PROPERTY(Relation, DETAIL_NAME);
    		if DetailBlock is not NULL and lower(DetailBlock) not like '%segment' then
    		  WriteBlockHeaderDelimited(pFileId,Delimiter,DetailBlock,TabPage,indent+1);
        end if;
    		Relation := GET_RELATION_PROPERTY(Relation, NEXT_MASTER_RELATION);
    	end loop;--Master-Detail
  	Text_IO.NEW_LINE(pFileId,1);
  END WriteBlockHeaderDelimited;
  
  --
  -- Output a block of data in delimited format.
  -- Master-Detail is handled using recursion.
  -- 
  PROCEDURE WriteBlockDataDelimited( pFileId    in Text_IO.file_type,
                                     Delimiter   IN VARCHAR2,
                                     BlockName   in VARCHAR2,
                                     TabPage     IN VARCHAR2,
                                     Depth       in NUMBER := 0)
  IS
  	v_count  		PLS_INTEGER  := 0;
  	block_id 		BLOCK;
  	start_rec   NUMBER := 1;
  	stop_rec		NUMBER;
  	ItemId      item;
  	item_count	NUMBER := 0;
  	first_name  VARCHAR2(65);
  	ItemName    VARCHAR2(65);
  	item_value  VARCHAR2(32767);
  	CurrentItem VARCHAR2(40);
  	ItemType    VARCHAR2(20);
  	SaveItem    VARCHAR2(40);
  	Relation    VARCHAR2(40);
  	DetailRelation VARCHAR2(40);
  	DetailBlock VARCHAR2(40);
  	ContextSaved BOOLEAN;
  BEGIN
--  	break;    
     if BlockName is NULL or
    	 Depth >= 5 or
    	 IsBlockVisible(BlockName) = FALSE or
    	 (BlockName not in ('COMPARE_RECORD_COUNT_BLK') 
    	 --and 
    	  --GET_BLOCK_PROPERTY(BlockName, QUERY_DATA_SOURCE_TYPE ) = 'NONE'
    	  ) then
      return;
    end if;
    if Depth = 0 then
      WriteBlockHeaderDelimited(pFileId,Delimiter, BlockName, TabPage, Depth);
    end if;
    block_id := FIND_BLOCK(BlockName);
--  	stop_rec := GET_BLOCK_PROPERTY(block_id, QUERY_HITS);
--  	IF stop_rec <= 0 then return; end if;
    Go_Block( BlockName );
    DetailRelation := GET_BLOCK_PROPERTY(block_id, FIRST_DETAIL_RELATION);
    if DetailRelation is not NULL then
    	Execute_Query(NO_VALIDATE); -- Fix SCR 0037 -- Madhu -- 31-Jan-2012
    end if;
    -- Output all records in the block
 		LAST_RECORD;
 		stop_rec := NAME_IN('SYSTEM.cursor_record'); -- actual last record
  	GO_RECORD(start_rec);
  	--break;
  	FOR i IN start_rec..stop_rec LOOP
    	item_count := 0;
  		first_name := GET_BLOCK_PROPERTY(block_id, FIRST_ITEM);
    	ItemName := BlockName||'.'||first_name;
    	ItemId := FIND_ITEM(ItemName);
    	-- Indentation
    	for j in 1..Depth loop
    	  Text_IO.PUT(pFileId, Delimiter); -- use delimiter to indent
      end loop;
    	WHILE ItemName IS NOT NULL LOOP
    	 	if IsDataItemVisible( ItemId, TabPage ) = TRUE then
    	    item_value := NAME_IN(ItemName);
    	    IF item_count > 0 THEN
    	      Text_IO.PUT(pFileId, DELIMITER);
    	    END IF;
  		    Text_IO.PUT(pFileId, '"'||item_value||'"');
  		    item_count := item_count + 1;
    	  end if; -- item visible
  		  ItemName := GET_ITEM_PROPERTY(ItemId, NEXTITEM);
  		  IF ItemName IS NOT NULL THEN
  			  ItemName := BlockName||'.'||ItemName;
  			  ItemId := FIND_ITEM(ItemName);
  		  ELSE
  			  Text_IO.NEW_LINE(pFileId, 1);
  		  END IF;
    	END LOOP; -- WHILE
    	-- Get the Master-Detail relationships for the record
    	Relation := GET_BLOCK_PROPERTY(block_id, FIRST_MASTER_RELATION);
--    	Relation := GET_BLOCK_PROPERTY(block_id, FIRST_DETAIL_RELATION);
      if Relation is not NULL then -- save context
      	ContextSaved := TRUE;
      else
      	ContextSaved := FALSE;
      end if;
    	while Relation is not NULL loop
    		DetailBlock := GET_RELATION_PROPERTY(Relation, DETAIL_NAME);
    		if DetailBlock is not NULL and lower(DetailBlock) not like '%segment' then
    		  WriteBlockDataDelimited(pFileId,Delimiter,DetailBlock,TabPage,Depth+1);
    		  if NAME_IN('System.Cursor_Block') <> BlockName then
  	  		  GO_BLOCK(BlockName); -- Restore block context
  	  	  end if;
        end if;
    		Relation := GET_RELATION_PROPERTY(Relation, NEXT_MASTER_RELATION);
    	end loop;--Master-Detail
    	if ContextSaved = TRUE then -- restore context
        Go_Block( BlockName );
    		Go_Record( i );
    	end if;
     	IF i < stop_rec THEN
    	  NEXT_RECORD;
    	ELSE
    	  EXIT;
    	END IF;
  	END LOOP; -- FOR start_rec > stop_rec
  	FIRST_RECORD;
  	RETURN;
  END WriteBlockDataDelimited;
